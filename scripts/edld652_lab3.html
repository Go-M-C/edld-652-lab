library(tidyverse)
library(here)
library(viridis)
df <- read.csv(here("data","gapminder.csv"))
  1. Create a bar chart
countinent_counts <- df %>% 
  group_by(continent) %>% 
  summarize(countries = n_distinct(country))


countinent_counts %>% 
  ggplot(aes(x = continent, y = countries, fill = continent))+
  geom_col() +
  labs(title = "Plot 1")+
  theme_minimal()

  1. Use the discrete viridis color palette instead to see if this is better
countinent_counts %>% 
  ggplot(aes(x = continent, y = countries, fill = continent))+
  geom_col() +
  labs(title = "Plot 2") +
  scale_fill_viridis_d() +
  theme_minimal()

  1. Use arguments in the scale function, for instance, use the plasma palette from the viridis package as an option Would you change anything in the colors you see to improve he visualization?
countinent_counts %>% 
  ggplot(aes(x = continent, y = countries, fill = continent))+
  geom_col() +
  labs(title = "Plot 3") +
  scale_fill_viridis_d(option = "plasma") +
  theme_minimal()

library(wesanderson)
